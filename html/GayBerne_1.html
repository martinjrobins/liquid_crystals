L = 50.0
k = 3.0
sigma_s = 1.0
rho = 0.7
rot_step = 2*pi/300
diff_step = sigma_s/50
dt = 0.004
T = 3.2
#vol = (4.0/3.0)*pi*k*sigma_s**3
area = (1.0/4.0)*pi*k*sigma_s**2
N = int(rho*L**2/area)

params = Params()
params['Dtrans'] = diff_step**2/(2.0*dt)*100
params['Drot'] = rot_step**2/(2.0*dt)*100
params['Temp'] = T
params['dt'] = dt
params['L'] = L

particles = Particles()
for i in range(N):
    p = Particle()
    eps = 0.0001
    p.position = Vect3d(uniform(eps,L-eps),uniform(eps,L-eps),0)
    theta = uniform(0,2*pi)
    u = [cos(theta),sin(theta),0]
    p.orientation = Vect3d(u)
    particles.append(p)
